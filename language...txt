#include <stdio.h>

int main() {
    int lang;
    printf("1, tamil\n");
    printf("2, english\n");
    printf("3, hindi\n");
    printf("4, malayalam\n");
    printf("5, kanada\n");
    printf("Enter your language: ");
    scanf("%d",&lang);

    switch (lang){
        case 1:
        printf("you have choose tamil\n");
        break;
        case 2:
        printf("you have choose english\n");
        break;
        case 3:
        printf("you have choose hindi\n");
        break;
        case 4:
        printf("you have choose malayalam\n");
        break;
        case 5:
        printf("you have choose kanada\n");
        break;
        default:
        printf("invalid input");
        
    }

    return 0;
}
